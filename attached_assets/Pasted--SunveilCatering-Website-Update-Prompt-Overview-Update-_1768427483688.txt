### Catering Product Card

```typescript
const CateringProductCard = ({ product }) => {
  const [quantity, setQuantity] = useState(product.cateringMinimum);
  
  return (
    <div className="catering-product-card">
      <img src={product.images.hero} alt={product.name} />
      <h4>{product.name}</h4>
      <p className="description">{product.description}</p>
      
      {/* Unit-aware pricing */}
      <div className="pricing">
        ${product.price} {product.unitLabel}
      </div>
      
      {/* Minimum order notice */}
      <div className="minimum-notice">
        Minimum: {product.cateringMinimum} {product.cateringUnit}
      </div>
      
      {/* Quantity selector */}
      <div className="quantity-selector">
        <button 
          onClick={() => setQuantity(Math.max(product.cateringMinimum, quantity - 1))}
          disabled={quantity <= product.cateringMinimum}
        >
          −
        </button>
        <span>{quantity} {product.cateringUnit}</span>
        <button onClick={() => setQuantity(quantity + 1)}>
          +
        </button>
      </div>
      
      {/* Line total */}
      <div className="line-total">
        Subtotal: ${quantity * product.price}
      </div>
      
      <button className="add-to-inquiry">
        Add to Inquiry
      </button>
    </div>
  );
};
```
