Database Schema Updates
sql-- Add to existing schema:

-- Catering/Wholesale Inquiries
CREATE TABLE inquiries (
    id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    inquiry_number NVARCHAR(20) UNIQUE NOT NULL,  -- CAT-2024-001234
    inquiry_type NVARCHAR(20) NOT NULL,  -- 'catering' or 'wholesale'
    status NVARCHAR(20) DEFAULT 'new',   -- new, contacted, quoted, converted, closed
    
    -- Contact Info
    contact_name NVARCHAR(255) NOT NULL,
    business_name NVARCHAR(255),
    email NVARCHAR(255) NOT NULL,
    phone NVARCHAR(20) NOT NULL,
    
    -- Event Details (catering)
    event_date DATE,
    event_type NVARCHAR(100),
    guest_count INT,
    event_location NVARCHAR(500),
    
    -- Business Details (wholesale)
    business_type NVARCHAR(100),
    delivery_frequency NVARCHAR(50),
    estimated_volume NVARCHAR(100),
    
    -- Common
    items NVARCHAR(MAX) NOT NULL,  -- JSON array of {productId, quantity, notes}
    additional_notes NVARCHAR(MAX),
    how_did_you_hear NVARCHAR(255),
    
    created_at DATETIME2 DEFAULT GETUTCDATE(),
    updated_at DATETIME2 DEFAULT GETUTCDATE(),
    responded_at DATETIME2,
    responded_by NVARCHAR(255)
);
